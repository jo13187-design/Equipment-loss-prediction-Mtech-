# Create region feature from coordinates
def get_region(lat, lon):
    # This is a simplified approach - in a real scenario you'd use shapefiles
    if pd.isna(lat) or pd.isna(lon):
        return 'Unknown'

    # Rough regional boundaries
    if lon > 34 and lat > 50: return 'Kyiv Oblast'
    elif lon > 36: return 'Kharkiv Oblast'
    elif lat < 47: return 'Kherson Oblast'
    elif lon < 30: return 'Chernihiv Oblast'
    else: return 'Other Oblast'

df['region'] = df.apply(lambda x: get_region(x['latitude'], x['longitude']), axis=1)

# Corrected objective function for PSO feature selection
def pso_feature_selection(X, y):
    def f_pso(feature_subset):
        # Convert float array to boolean mask
        feature_subset = feature_subset > 0.5  # Using 0.5 as threshold

        # Handle case where no features are selected
        if np.sum(feature_subset) == 0:
            return 999  # Return high error

        # Train model with selected features
        model = RandomForestRegressor(n_estimators=50, random_state=42)
        scores = []

        # Time-series cross-validation
        tscv = TimeSeriesSplit(n_splits=3)
        for train_idx, test_idx in tscv.split(X):
            X_train, X_test = X[train_idx], X[test_idx]
            y_train, y_test = y.iloc[train_idx], y.iloc[test_idx]

            # Corrected indexing - use boolean mask directly
            model.fit(X_train[:, feature_subset], y_train)
            y_pred = model.predict(X_test[:, feature_subset])
            scores.append(mean_squared_error(y_test, y_pred))

        return np.mean(scores)

    return f_pso

# Initialize PSO

# Prepare data for LSTM
def create_sequences(data, seq_length):
    X, y = [], []
    for i in range(len(data) - seq_length):
        X.append(data[i:i+seq_length])
        y.append(data[i+seq_length])
    return np.array(X), np.array(y)

# Normalize and create sequences

# Hyperparameter Optimization with Differential Evolution (DE)

# Define objective function for Random Forest hyperparameters
def rf_objective(params):
    n_estimators = int(params[0])
    max_depth = int(params[1]) if params[1] > 0 else None
    min_samples_split = int(params[2])

    model = RandomForestRegressor(
        n_estimators=n_estimators,
        max_depth=max_depth,
        min_samples_split=min_samples_split,
        random_state=42
    )

    # Time-series cross-validation
    tscv = TimeSeriesSplit(n_splits=3)
    scores = []
    for train_idx, test_idx in tscv.split(X_train_scaled):
        X_train, X_test = X_train_scaled[train_idx], X_train_scaled[test_idx]
        y_train, y_test = y_train_reg.iloc[train_idx], y_train_reg.iloc[test_idx]

        model.fit(X_train, y_train)
        y_pred = model.predict(X_test)
        scores.append(mean_squared_error(y_test, y_pred))

    return np.mean(scores)

# Define bounds for parameters
bounds = [(10, 200), (2, 50), (2, 20)]  # n_estimators, max_depth, min_samples_split

# Run DE optimization
print("Running Differential Evolution Optimization...")
de = DE(func=rf_objective, n_dim=3, size_pop=10, max_iter=20, lb=[b[0] for b in bounds], ub=[b[1] for b in bounds])
best_params, best_score = de.run()

print(f"\nBest parameters: n_estimators={int(best_params[0])}, max_depth={int(best_params[1])}, min_samples_split={int(best_params[2])}")
print(f"Best MSE: {best_score:.4f}")

# Save optimization results
optimization_results = pd.DataFrame({
    'Parameter': ['n_estimators', 'max_depth', 'min_samples_split'],
    'Best_Value': [int(best_params[0]), int(best_params[1]), int(best_params[2])],
    'Best_MSE': [best_score]
})
